/* ---------- 공용 콘솔 함수 ---------- */

void gotoxy(int x,int y){
    COORD pos={x,y};
    SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE),pos);
}

/* ---------- 빠른 단어 치기 ---------- */

typedef struct {
    int active;
    int x, y;
    char word[100];
} FallingWord;

int hasActiveWords(FallingWord words[], int maxWords) {
    int i;
    for (i = 0; i < maxWords; i++)
        if (words[i].active) return 1;
    return 0;
}

/* 스테이지에 따라 단어 길이/리스트 선택 */
void getRandomWordForStage(int st, char *target) {
    int minLen;
    if (st <= 5) minLen = 0;
    else if (st <= 10) minLen = 6;
    else minLen = 10;

    while (1) {
        int r = rand() % 30;
        int z;
        char temp[100];

        if (st <= 5) strcpy(temp, list1[r]);
        else if (st <= 10) {
            if (rand() % 2) strcpy(temp, list2[r]);
            else strcpy(temp, list3[r]);
        } else {
            z = rand() % 3;
            if (z == 0) strcpy(temp, list4[r]);
            else if (z == 1) strcpy(temp, list5[r]);
            else strcpy(temp, list6[r]);
        }

        if ((int)strlen(temp) >= minLen) {
            strcpy(target, temp);
            return;
        }
    }
}

/* 단어가 겹치지 않도록 x좌표를 찾는 함수 */
int findSafeX(FallingWord words[], int maxWords, const char *newWord) {
    int lenNew = (int)strlen(newWord);
    int minX, maxX;
    int tryCount = 0;
    int x, i, lenOld;

    if (lenNew < 1) lenNew = 1;

    minX = PLAY_LEFT;
    maxX = PLAY_RIGHT - lenNew;
    if (maxX < minX) maxX = minX;

    while (1) {
        int ok = 1;
        x = minX;
        if (maxX > minX)
            x = minX + (rand() % (maxX - minX + 1));

        for (i = 0; i < maxWords; i++) {
            if (!words[i].active) continue;
            if (words[i].y != 4) continue;

            lenOld = (int)strlen(words[i].word);
            if (lenOld < 1) lenOld = 1;

            if (!(x + lenNew <= words[i].x || words[i].x + lenOld <= x)) {
                ok = 0;
                break;
            }
        }

        if (ok) return x;

        tryCount++;
        if (tryCount > 100) {
            return x;
        }
    }
}

void gameTyping() {
    int st;
    char input[100];
    int speed;
    int i;

    score = 0;
    life = 5;

    for (st = 1; st <= 15 && life > 0; st++) {
        int totalWords = 13 + (st * 2);
        int remaining = totalWords;

        int maxWords;
        if (st <= 3) maxWords = 2;
        else if (st <= 6) maxWords = 3;
        else if (st <= 10) maxWords = 4;
        else maxWords = 5;

        speed = 800 - st * 40;
        if (speed < 150) speed = 150;

        FallingWord words[5];

        for (i = 0; i < 5; i++) words[i].active = 0;

        while ((remaining > 0 || hasActiveWords(words, maxWords)) && life > 0) {

            /* 새 단어 생성 (겹치지 않게 x좌표 배치) */
            for (i = 0; i < maxWords && remaining > 0; i++) {
                if (!words[i].active) {
                    words[i].active = 1;
                    words[i].y = 4;
                    getRandomWordForStage(st, words[i].word);
                    words[i].x = findSafeX(words, maxWords, words[i].word);
                    remaining--;
                }
            }

            /* 화면 전체 지우기 */
            system("cls");

            gotoxy(0, 0);
            printf("====================================================\n");
            printf("  [빠른 단어 치기] 게임\n");
            printf(" STAGE: %d  |  SCORE: %d  |  LIFE: %d\n", st, score, life);
            printf(" 남은 생성 단어: %d\n", remaining);
            printf(" (단어 영역: %d ~ %d 열)\n", PLAY_LEFT, PLAY_RIGHT);
            printf("====================================================\n");

            for (i = 0; i < maxWords; i++) {
                if (words[i].active) {
                    gotoxy(words[i].x, words[i].y);
                    printf("%s", words[i].word);
                }
            }

            gotoxy(2, 22); printf("입력:                                ");
            gotoxy(8, 22);

            if (kbhit()) {
                int k = 0;

                while (1) {
                    int c = getch();
                    if (c == 13) {
                        input[k] = '\0';
                        break;
                    } else if (c == 8) {
                        if (k > 0) {
                            if (input[k - 1] < 0 && k >= 2) {
                                printf("\b\b  \b\b");
                                k -= 2;
                            } else {
                                printf("\b \b");
                                k--;
                            }
                        }
                    } else if (c == 224 || c == 0) {
                        getch();
                    } else {
                        if (k < 80) {
                            input[k++] = (char)c;
                            if (c & 0x80) {
                                int c2 = getch();
                                input[k++] = (char)c2;
                                printf("%c%c", c, c2);
                            } else {
                                printf("%c", c);
                            }
                        } else if (c & 0x80) {
                            getch();
                        }
                    }
                }

                {
                    int matchedIndex = -1;
                    for (i = 0; i < maxWords; i++) {
                        if (words[i].active && strcmp(input, words[i].word) == 0) {
                            matchedIndex = i;
                            break;
                        }
                    }

                    if (matchedIndex >= 0) {
                        score += 10;
                        words[matchedIndex].active = 0;
                    } else {
                        life--;
                    }
                }

                if (life <= 0) break;
            }

            for (i = 0; i < maxWords; i++) {
                if (!words[i].active) continue;
                words[i].y++;
                if (words[i].y >= 20) {
                    words[i].active = 0;
                    life--;
                    if (life <= 0) break;
                }
            }

            if (life <= 0) break;

            Sleep(speed);
        }
    }

    system("cls");
    if (life <= 0) {
        gotoxy(10, 10);
        printf("게임 오버! 최종 점수: %d\n", score);
    } else {
        printf("\n\n  축하합니다! 모든 스테이지 클리어!\n");
        printf("  최종 점수: %d\n", score);
    }
    getch();
}

/* ---------- 벽돌깨기 ---------- */

#define BB_WIDTH  60
#define BB_HEIGHT 25
#define BB_PADDLE_WIDTH 9
#define BB_MAX_BRICKS 40   /* 최대 벽돌 개수 */

void drawBBBorder() {
    int x, y;
    for (x = 1; x <= BB_WIDTH; x++) {
        gotoxy(x, 3);          printf("-");
        gotoxy(x, BB_HEIGHT);  printf("-");
    }
    for (y = 3; y <= BB_HEIGHT; y++) {
        gotoxy(1, y);         printf("|");
        gotoxy(BB_WIDTH, y);  printf("|");
    }
}

void gameBrickBreaker() {
    int brickX[BB_MAX_BRICKS];
    int brickY[BB_MAX_BRICKS];
    int brickActive[BB_MAX_BRICKS];

    int paddleX, paddleY;
    int ballX, ballY, dirX, dirY;
    int bbScore = 0;
    int bbLife  = 3;

    int stage = 1;
    int maxStage = 10;

    while (bbLife > 0 && stage <= maxStage) {
        int brickCount = 7 + (stage - 1) * 3;  /* 7개부터 시작해서 점점 증가 */
        int i, bricksLeft;
        int prevBallX, prevBallY, prevPaddleX;

        if (brickCount > BB_MAX_BRICKS) brickCount = BB_MAX_BRICKS;

        /* 스테이지 초기화 */
        system("cls");
        printf("=============== [벽돌깨기] ===============\n");
        printf(" STAGE: %d   점수: %d   목숨: %d\n", stage, bbScore, bbLife);
        printf(" 이동: 왼쪽/오른쪽 화살표 또는 A/D  |  종료: Q\n");

        drawBBBorder();

        /* 벽돌 위치 배치 (위에서부터 차곡차곡) */
        for (i = 0; i < brickCount; i++) {
            int row = i / 10;          /* 한 줄에 최대 10개 */
            int col = i % 10;
            brickActive[i] = 1;
            brickX[i] = 4 + col * 5;   /* 4,9,14,... */
            brickY[i] = 5 + row;       /* 5,6,7,... */
        }

        for (i = brickCount; i < BB_MAX_BRICKS; i++) brickActive[i] = 0;

        /* 초기 패들, 공 위치 */
        paddleX = (BB_WIDTH - BB_PADDLE_WIDTH) / 2;
        paddleY = BB_HEIGHT - 2;

        ballX = BB_WIDTH / 2;
        ballY = paddleY - 1;
        dirX = (rand() % 2) ? 1 : -1;
        dirY = -1;

        prevBallX = ballX;
        prevBallY = ballY;
        prevPaddleX = paddleX;

        /* 벽돌 그리기(한 번만) */
        for (i = 0; i < brickCount; i++) {
            if (brickActive[i]) {
                gotoxy(brickX[i], brickY[i]);
                printf("####");
            }
        }

        /* 패들, 공 초기 출력 */
        gotoxy(paddleX, paddleY);
        for (i = 0; i < BB_PADDLE_WIDTH; i++) printf("=");

        gotoxy(ballX, ballY);
        printf("O");

        while (1) {
            /* 상단 정보는 매 프레임 업데이트 */
            gotoxy(0, 0);
            printf("=============== [벽돌깨기] ===============\n");
            printf(" STAGE: %d   점수: %d   목숨: %d                     \n", stage, bbScore, bbLife);
            printf(" 이동: 왼쪽/오른쪽 화살표 또는 A/D  |  종료: Q       \n");

            /* 입력 처리 */
            if (kbhit()) {
                int c = getch();
                if (c == 'a' || c == 'A') {
                    paddleX -= 2;
                } else if (c == 'd' || c == 'D') {
                    paddleX += 2;
                } else if (c == 'q' || c == 'Q') {
                    return;
                } else if (c == 224 || c == 0) {
                    int c2 = getch();
                    if (c2 == 75) paddleX -= 2;
                    else if (c2 == 77) paddleX += 2;
                }
            }

            if (paddleX < 2) paddleX = 2;
            if (paddleX + BB_PADDLE_WIDTH >= BB_WIDTH)
                paddleX = BB_WIDTH - BB_PADDLE_WIDTH - 1;

            /* 이전 패들/공 지우기 */
            gotoxy(prevPaddleX, paddleY);
            for (i = 0; i < BB_PADDLE_WIDTH; i++) printf(" ");

            gotoxy(prevBallX, prevBallY);
            printf(" ");

            /* 공 이동 */
            prevBallX = ballX;
            prevBallY = ballY;
            prevPaddleX = paddleX;

            ballX += dirX;
            ballY += dirY;

            /* 벽 충돌 */
            if (ballX <= 2) {
                ballX = 2;
                dirX = 1;
            } else if (ballX >= BB_WIDTH - 1) {
                ballX = BB_WIDTH - 1;
                dirX = -1;
            }
            if (ballY <= 4) {
                ballY = 4;
                dirY = 1;
            }

            /* 패들 충돌 */
            if (ballY == paddleY - 1 &&
                ballX >= paddleX &&
                ballX <= paddleX + BB_PADDLE_WIDTH) {
                dirY = -1;
                if (ballX < paddleX + BB_PADDLE_WIDTH / 2) dirX = -1;
                else dirX = 1;
            }

            /* 벽돌 충돌 */
            for (i = 0; i < brickCount; i++) {
                if (!brickActive[i]) continue;
                if (ballY == brickY[i] &&
                    ballX >= brickX[i] &&
                    ballX <  brickX[i] + 4) {
                    brickActive[i] = 0;
                    dirY *= -1;
                    bbScore += 10;

                    /* 부서진 벽돌 지우기 */
                    gotoxy(brickX[i], brickY[i]);
                    printf("    ");
                    break;
                }
            }

            /* 패들 다시 그리기 */
            gotoxy(paddleX, paddleY);
            for (i = 0; i < BB_PADDLE_WIDTH; i++) printf("=");

            /* 공 그리기 */
            gotoxy(ballX, ballY);
            printf("O");

            /* 벽돌 남았는지 체크 */
            bricksLeft = 0;
            for (i = 0; i < brickCount; i++)
                if (brickActive[i]) bricksLeft++;

            if (bricksLeft == 0) {
                gotoxy(10, BB_HEIGHT - 1);
                printf("STAGE %d CLEAR! 아무 키나 누르세요...", stage);
                getch();
                stage++;
                break;
            }

            /* 공이 바닥 밑으로 떨어지면 목숨 감소 */
            if (ballY >= BB_HEIGHT - 1) {
                bbLife--;
                if (bbLife <= 0) break;

                gotoxy(ballX, ballY); printf(" ");
                ballX = BB_WIDTH / 2;
                ballY = paddleY - 1;
                dirX = (rand() % 2) ? 1 : -1;
                dirY = -1;
                prevBallX = ballX;
                prevBallY = ballY;

                gotoxy(10, BB_HEIGHT - 1);
                printf("남은 목숨: %d  (아무 키나 누르세요...)", bbLife);
                getch();
                gotoxy(10, BB_HEIGHT - 1);
                printf("                                          ");
            }

            Sleep(45);
        }

        if (bbLife <= 0) break;
    }

    system("cls");
    gotoxy(10, 10);
    if (bbLife <= 0)
        printf("벽돌깨기 게임 오버! 점수: %d\n", bbScore);
    else
        printf("모든 스테이지 클리어! 점수: %d\n", bbScore);
    getch();
}

/* ---------- 메뉴 ---------- */

void showMenu() {
    int choice;

    while (1) {
        system("cls");
        printf("=====================================\n");
        printf("        타자 / 벽돌깨기 게임\n");
        printf("=====================================\n");
        printf(" 1. 빠른 단어 치기\n");
        printf(" 2. 벽돌깨기\n");
        printf(" 0. 종료\n");
        printf("=====================================\n");
        printf(" 선택: ");

        if (scanf("%d", &choice) != 1) {
            int ch;
            while ((ch = getchar()) != '\n' && ch != EOF);
            continue;
        }
        {
            int ch;
            while ((ch = getchar()) != '\n' && ch != EOF);
        }

        if (choice == 0) break;
        else if (choice == 1) {
            gameTyping();
        } else if (choice == 2) {
            gameBrickBreaker();
        } else {
            printf("잘못된 선택입니다. 아무 키나 누르세요.\n");
            getch();
        }
    }
}

int main() {
    srand((unsigned)time(NULL));
    showMenu();
    return 0;
}
